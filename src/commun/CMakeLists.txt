function(include_file path)
    file(READ ${path} content)
    set(delim "for_c++_include")
    set(content "R\"${delim}(\n${content})${delim}\"")
    file(APPEND "embedded" "{ \"${path}\", ${content} },\n")
endfunction(include_file)
